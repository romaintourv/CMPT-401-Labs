import time
from machine import Pin

led = Pin(2, Pin.OUT)
button1 = Pin(13, Pin.IN,Pin.PULL_UP)
button2 = Pin(12, Pin.IN,Pin.PULL_UP)
lastPressed = None


def reverseGPIO():
if led.value():
led.value(0)
else:
led.value(1)

while True:
if not button1.value():
time.sleep_ms(20)
if not button1.value() and lastPressed != 1:
lastPressed = 1
reverseGPIO()

while not button2.value():
time.sleep_ms(20)

if not button2.value():
time.sleep_ms(20)
if not button2.value() and lastPressed != 2:
lastPressed = 2
reverseGPIO()

while not button1.value():
time.sleep_ms(20)
